jQuery utility functions

Array methods

isArray

Is an item an array?

var a = [1,2,3];
console.log( $.isArray(a) ); // true

inArray

Is an item in an array?

var a = [1,2,3];
console.log( $.inArray(1, a) ); // 0
console.log( $.inArray(3, a) ); // 2
console.log( $.inArray(4, a) ); // -1

makeArray

Create a real array from an array-like object.

var p = document.getElementsByTagName("p");
var pArray = $.makeArray(p);

merge

Merge the contents of two arrays into the first array retaining order and duplicate values.

var a = [0, 1, 2];
var b = [2, 3, 4];
$.merge(a, b);
console.log( a );
[0, 1, 2, 2, 3, 4]

extend

Merges the contents of two objects/arrays and modifies the first. If duplicates are found, the second takes priority.

var a = { alpha: 1, beta: 2, gamma: 3 };
var b = { alpha: 0, delta: 999 };
$.extend(a, b);
console.log( a );
{ alpha: 0, beta: 2, gamma: 3, delta: 999 };

each

Can iterate over a jQuery object:

$("p").each( function(i, e) {
	console.log( "paragraph ", i, ": ", e );
});

or an array

$.each( [1,2,3], function(i, e) {
	console.log( "item at index ", i, " is ", e );
});

map

Translate all items in an array/object to a new array/object.

var a = ['a', 'b', 'c'];
var b = $.map(a, function(i, e){
	return e+i;
});

console.log( b );
['a0', 'b1', 'c2']
